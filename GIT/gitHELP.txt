git status - были ли изменения в репозитории
pwd - узнать путь
ls - список файлов
cd ~/[имя папки] - перейти в каталог
touch [имя файла] - создание файла
mkdir [имя папки] - создание папки
echo 'сообщение' - вывод сообщения в консоль
echo 'сообщение' > [имя файла] - перезапись файла
echo 'сообщение' >> [имя файла] - дозапись файла
cat [имя файла] - прочитать файл
unlink [имя файла] - удаление файла
git diff - какие изменения внесены
git diff --cached - какие изменения проиндексированных файлов
git checkout -- [имя файла] - откатить изменения файла
git reset HEAD [имя файла] - перевод в неиндексируемое состояние
git log - история изменений
git log -p - подробрная история измненений 
git blame [имя файла] - кто какое изменение в файле делал

git branch -m [название] [новое название] - переименовать ветку
git merge [название] - слияние веток, нужно находиться на главной

СОЗДАНИЕ, ВЫГРУЗКА, ЗАГРУЗКА РЕПОЗИТОРИЯ GITHUB
Создаем новый репозиторий на сайте
Выполняем команды по подключению готового репозитория
git push -u origin master - отправляем на сервер репозиторий
git remote - список удаленнных репозиториев
git clone [ssh с репозитория] - для скачивания репозитория с github
git pull origin master - скачать изменения с github
git fetch - получение всех изменений по всем веткам
git branch -r - список веток на github
git push origin [имя ветки] - отправка ветки на github
git push origin :[имя ветки] - удаление ветки на github
git branch [имя новой] origin/[имя копируемой] - создание локальной копии
git stash - хранение проиндексированных изменений в stash (при этом коммита нет)
git stash list - список stash
git stash apply - вернуть в работе незакомиченные изменения
git stash drop - очистка stash
git tag [новое имя] - создание нового тега
git show - полное описание текущего коммита
git show [тег/хэш] - просмотр указанного коммита
git show HEAD - просмотр последнего коммита
git tag -a [имя тега] -m 'описание' [хэш коммита] - тег для указанного коммита
git checkout -b [имя новой ветки] [тэг коммита] - создать ветку от определенного коммита
git push origin [тег] - отправка тега на github
git push origin --tags - отправка всех тегов на github
git ls-remote --tags origin - список тегов на github
git push origin :[тег] - удаление тега на github
